   109→// ─── Helpers ────────────────────────────────────────────────────────────────
   110→
   111→function getPainLimit(serviceCount: number): number {
   112→  if (serviceCount === 1) return 2
   113→  if (serviceCount === 2) return 3
   114→  return 4
   115→}
   116→
   117→function getAvailablePains(services: string[]): string[] {
   118→  const pains = new Set<string>()
   119→  for (const svc of services) {
   120→    const entry = SERVICE_PAIN_MAP[svc]
   121→    if (entry) entry.painPoints.forEach((p) => pains.add(p))
   122→  }
   123→  return [...pains]
   124→}
   125→
   126→function getAvailableProjects(services: string[]): string[] {
   127→  const projects = new Set<string>()
   128→  for (const svc of services) {
   129→    const entry = SERVICE_PAIN_MAP[svc]
   130→    if (entry) entry.projectDetails.forEach((p) => projects.add(p))
   131→  }
   132→  return [...projects]
   133→}

   135→// ─── Step Transition Wrapper ────────────────────────────────────────────────
   136→
   137→const STEP_VARIANTS = {
   138→  enter: (dir: number) => ({
   139→    x: dir > 0 ? 80 : -80,
   140→    opacity: 0,
   141→  }),
   142→  center: { x: 0, opacity: 1 },
   143→  exit: (dir: number) => ({
   144→    x: dir > 0 ? -80 : 80,
   145→    opacity: 0,
   146→  }),
   147→}

   149→// ─── Main Component ─────────────────────────────────────────────────────────

   151→export default function GetStartedPage() {
   152→  const pageRef = useRef<HTMLDivElement>(null)
   153→  const mousePosition = useMousePosition(pageRef)
   154→
   155→  const [state, setState] = useState<QuestionnaireState>(INITIAL_STATE)
   156→  const [direction, setDirection] = useState(1)
   157→
   158→  const update = useCallback(<K extends keyof QuestionnaireState>(key: K, value: QuestionnaireState[K]) => {
   159→    setState((prev) => ({ ...prev, [key]: value }))
   160→  }, [])
   161→
   162→  const goNext = useCallback(() => {
   163→    setDirection(1)
   164→    setState((prev) => ({ ...prev, step: Math.min(prev.step + 1, 7) as Step }))
   165→  }, [])
   166→
   167→  const goBack = useCallback(() => {
   168→    setDirection(-1)
   169→    setState((prev) => ({ ...prev, step: Math.max(prev.step - 1, 1) as Step }))
   170→  }, [])
   171→
   172→  const goToStep = useCallback((step: Step) => {
   173→    setDirection(step > state.step ? 1 : -1)
   174→    setState((prev) => ({ ...prev, step }))
   175→  }, [state.step])
   176→
   177→  const toggleService = useCallback((svc: string) => {
   178→    setState((prev) => {
   179→      const has = prev.selectedServices.includes(svc)
   180→      const next = has
   181→        ? prev.selectedServices.filter((s) => s !== svc)
   182→        : [...prev.selectedServices, svc]
   183→      // Reset downstream state
   184→      return { ...prev, selectedServices: next, selectedPains: [], painCustomText: "", selectedProjects: [], projectCustomText: "" }
   185→    })
   186→  }, [])
   187→
   188→  const togglePain = useCallback((pain: string) => {
   189→    setState((prev) => {
   190→      if (pain === "__custom__") {
   191→        const has = prev.selectedPains.includes("__custom__")
   192→        if (!has && prev.selectedPains.length >= getPainLimit(prev.selectedServices.length)) return prev
   193→        const next = has
   194→          ? prev.selectedPains.filter((p) => p !== "__custom__")
   195→          : [...prev.selectedPains, "__custom__"]
   196→        return { ...prev, selectedPains: next }
   197→      }
   198→      const has = prev.selectedPains.includes(pain)
   199→      if (!has && prev.selectedPains.length >= getPainLimit(prev.selectedServices.length)) return prev
   200→      const next = has
   201→        ? prev.selectedPains.filter((p) => p !== pain)
   202→        : [...prev.selectedPains, pain]
   203→      return { ...prev, selectedPains: next }
   204→    })
   205→  }, [])
   206→
   207→  const toggleProject = useCallback((proj: string) => {
   208→    setState((prev) => {
   209→      if (proj === "__custom_project__") {
   210→        const has = prev.selectedProjects.includes("__custom_project__")
   211→        if (!has && prev.selectedProjects.length >= 3) return prev
   212→        const next = has
   213→          ? prev.selectedProjects.filter((p) => p !== "__custom_project__")
   214→          : [...prev.selectedProjects, "__custom_project__"]
   215→        return { ...prev, selectedProjects: next }
   216→      }
   217→      const has = prev.selectedProjects.includes(proj)
   218→      if (!has && prev.selectedProjects.length >= 3) return prev
   219→      const next = has
   220→        ? prev.selectedProjects.filter((p) => p !== proj)
   221→        : [...prev.selectedProjects, proj]
   222→      return { ...prev, selectedProjects: next }
   223→    })
   224→  }, [])
   225→
   226→  const toggleContactPref = useCallback((pref: string) => {
   227→    setState((prev) => {
   228→      const has = prev.contactPrefs.includes(pref)
   229→      return { ...prev, contactPrefs: has ? prev.contactPrefs.filter((p) => p !== pref) : [...prev.contactPrefs, pref] }
   230→    })
   231→  }, [])
   232→
   233→  // Derived values
   234→  const painLimit = getPainLimit(state.selectedServices.length)
   235→  const availablePains = useMemo(() => getAvailablePains(state.selectedServices), [state.selectedServices])
   236→  const availableProjects = useMemo(() => getAvailableProjects(state.selectedServices), [state.selectedServices])
   237→  const selectedBudget = BUDGET_OPTIONS.find((b) => b.value === state.budget)
   238→  const totalSteps = 6
   239→  const progressPct = state.step <= 6 ? (state.step / totalSteps) * 100 : 100
   240→
   241→  // Validation helpers
   242→  const canStep1 = state.selectedServices.length >= 1
   243→  const canStep2 = state.selectedPains.length >= 1
   244→  const canStep3 = state.selectedProjects.length >= 1
   245→  const canStep4 = state.firstName.trim() !== "" && state.email.trim() !== "" && state.contactPrefs.length >= 1
   246→  const canStep5 = state.budget !== ""
   247→
   248→  const canProceed = state.step === 1 ? canStep1
   249→    : state.step === 2 ? canStep2
   250→    : state.step === 3 ? canStep3
   251→    : state.step === 4 ? canStep4
   252→    : state.step === 5 ? canStep5
   253→    : true
   254→
   255→  // ─── Render ─────────────────────────────────────────────────────────────────
   256→
   257→  return (
   258→    <main ref={pageRef} className="relative min-h-screen bg-white">
   259→      {/* Step 7 uses its own dark background, steps 1-6 use circuitry */}
   260→      {state.step <= 6 && (
   261→        <CircuitryBackground mousePosition={mousePosition} />
   262→      )}
   263→
   264→      {/* ── Progress Bar (steps 1-6 only) ── */}
   265→      {state.step <= 6 && (
   266→        <div className="fixed top-0 left-0 right-0 z-50 h-[2px] bg-gray-100">
   267→          <motion.div
   268→            className="h-full bg-[#111111]"
   269→            initial={false}
   270→            animate={{ width: `${progressPct}%` }}
   271→            transition={{ duration: 0.4, ease: EASE }}
   272→          />
   273→        </div>
   274→      )}
   275→
   276→      {/* ── Hero (compact, steps 1-6 only) ── */}
   277→      {state.step <= 6 && (
   278→        <section className="relative min-h-[40vh] flex items-center justify-center px-4 pt-8 pb-4">
   279→          <div className="text-center max-w-3xl mx-auto relative z-10">
   280→            <motion.span
   281→              className="inline-block px-4 py-1.5 bg-gray-100 text-gray-600 text-xs font-medium tracking-wider uppercase rounded-full mb-6 border border-gray-200"
   282→              initial={{ scale: 0.8, opacity: 0 }}
   283→              animate={{ scale: 1, opacity: 1 }}
   284→              transition={{ duration: 0.5, ease: EASE }}
   285→            >
   286→              Get Started
   287→            </motion.span>
   288→            <motion.h1
   289→              className="text-4xl md:text-6xl font-bold tracking-tight leading-tight text-gray-900"
   290→              style={{ fontFamily: "var(--font-space-grotesk)" }}
   291→              initial={{ y: 30, opacity: 0 }}
   292→              animate={{ y: 0, opacity: 1 }}
   293→              transition={{ duration: 0.7, ease: EASE }}
   294→            >
   295→              Let&apos;s Build Something
   296→            </motion.h1>
   297→            <motion.p
   298→              className="mt-6 text-lg text-gray-600 max-w-2xl mx-auto leading-relaxed"
   299→              initial={{ y: 20, opacity: 0 }}
   300→              animate={{ y: 0, opacity: 1 }}
   301→              transition={{ duration: 0.7, delay: 0.2, ease: EASE }}
   302→            >
   303→              A few quick questions so we can hit the ground running. Takes about 2 minutes.
   304→            </motion.p>
   305→          </div>
   306→        </section>
   307→      )}
   308→
   309→      {/* ── Step 7: Full dark receipt page ── */}
   310→      {state.step === 7 && (
   311→        <Step7 firstName={state.firstName} />
   312→      )}
   313→
   314→      {/* ── Steps 1-6: Questionnaire Area ── */}
   315→      {state.step <= 6 && (
   316→        <section className="relative z-10 pb-32 md:pb-20">
   317→          <div className="max-w-3xl mx-auto px-4">
   318→            <AnimatePresence mode="wait" custom={direction}>
   319→              <motion.div
   320→                key={state.step}
   321→                custom={direction}
   322→                variants={STEP_VARIANTS}
   323→                initial="enter"
   324→                animate="center"
   325→                exit="exit"
   326→                transition={{ duration: 0.4, ease: EASE }}
   327→              >
   328→                {state.step === 1 && (
   329→                  <Step1
   330→                    selectedServices={state.selectedServices}
   331→                    toggleService={toggleService}
   332→                  />
   333→                )}
   334→                {state.step === 2 && (
   335→                  <Step2
   336→                    selectedPains={state.selectedPains}
   337→                    painCustomText={state.painCustomText}
   338→                    painLimit={painLimit}
   339→                    availablePains={availablePains}
   340→                    togglePain={togglePain}
   341→                    updatePainCustom={(v) => update("painCustomText", v)}
   342→                  />
   343→                )}
   344→                {state.step === 3 && (
   345→                  <Step3
   346→                    selectedProjects={state.selectedProjects}
   347→                    projectCustomText={state.projectCustomText}
   348→                    availableProjects={availableProjects}
   349→                    toggleProject={toggleProject}
   350→                    updateProjectCustom={(v) => update("projectCustomText", v)}
   351→                  />
   352→                )}
   353→                {state.step === 4 && (
   354→                  <Step4
   355→                    state={state}
   356→                    update={update}
   357→                    toggleContactPref={toggleContactPref}
   358→                  />
   359→                )}
   360→                {state.step === 5 && (
   361→                  <Step5
   362→                    budget={state.budget}
   363→                    selectBudget={(v) => update("budget", v)}
   364→                  />
   365→                )}
   366→                {state.step === 6 && (
   367→                  <Step6
   368→                    state={state}
   369→                    selectedBudget={selectedBudget}
   370→                    goToStep={goToStep}
   371→                    goNext={goNext}
   372→                  />
   373→                )}
   374→              </motion.div>
   375→            </AnimatePresence>
   376→          </div>
   377→
   378→          {/* ── Desktop Navigation (bottom) ── */}
   379→          {state.step >= 1 && state.step <= 5 && (
   380→            <div className="hidden md:flex items-center justify-between max-w-3xl mx-auto px-4 mt-8">
   381→              <button
   382→                onClick={goBack}
   383→                className="text-gray-500 hover:text-gray-900 transition-colors text-sm"
   384→                suppressHydrationWarning
   385→              >
   386→                ← Back
   387→              </button>
   388→              <button
   389→                onClick={goNext}
   390→                disabled={!canProceed}
   391→                className={`animate-cta-pulse rounded-full px-8 py-3 text-sm font-medium transition-colors ${
   392→                  canProceed
   393→                    ? "bg-gray-900 text-white hover:bg-gray-800 cursor-pointer"
   394→                    : "bg-gray-200 text-gray-400 cursor-not-allowed"
   395→                }`}
   396→                suppressHydrationWarning
   397→              >
   398→                Next →
   399→              </button>
   400→            </div>
   401→          )}
   402→
   403→          {/* ── Mobile Navigation (fixed bottom bar) ── */}
   404→          {state.step >= 1 && state.step <= 5 && (
   405→            <div className="md:hidden fixed bottom-0 left-0 right-0 z-50 bg-white/90 backdrop-blur-md border-t border-gray-200 px-4 py-3 flex items-center justify-between">
   406→              <button
   407→                onClick={goBack}
   408→                className="text-gray-500 hover:text-gray-900 transition-colors text-sm py-2 px-4"
   409→                suppressHydrationWarning
   410→              >
   411→                ← Back
   412→              </button>
   413→              <button
   414→                onClick={goNext}
   415→                disabled={!canProceed}
   416→                className={`animate-cta-pulse rounded-full px-6 py-2.5 text-sm font-medium transition-colors ${
   417→                  canProceed
   418→                    ? "bg-gray-900 text-white hover:bg-gray-800 cursor-pointer"
   419→                    : "bg-gray-200 text-gray-400 cursor-not-allowed"
   420→                }`}
   421→                suppressHydrationWarning
   422→              >
   423→                Next →
   424→              </button>
   425→            </div>
   426→          )}
   427→        </section>
   428→      )}
   429→    </main>
   430→  )
   431→}

   436→  selectedServices,
   437→  toggleService,
   438→}: {
   439→  selectedServices: string[]
   440→  toggleService: (s: string) => void
   441→}) {
   442→  return (
   443→    <div>
   444→      <h2
   445→        className="text-2xl md:text-3xl font-bold text-gray-900 tracking-tight"

   454→          animate={{ opacity: 1, y: 0 }}
   455→          className="text-gray-400 text-xs mb-4"
   456→        >
   457→          You picked {selectedServices.length} service{selectedServices.length !== 1 ? "s" : ""}

   461→      <div className="grid grid-cols-1 sm:grid-cols-2 gap-3 mt-6">
   462→        {Object.entries(SERVICE_PAIN_MAP).map(([key, svc]) => {
   463→          const selected = selectedServices.includes(key)
   464→          return (
   465→            <motion.button
   466→              key={key}
   467→              onClick={() => toggleService(key)}
   468→              className={`text-left px-6 py-4 rounded-xl border cursor-pointer transition-all duration-300 ${
   469→                selected
   470→                  ? "border-gray-900 bg-gray-50 shadow-sm"
   471→                  : "bg-white border-gray-200 hover:border-gray-400 hover:shadow-md"
   472→              }`}
   473→              whileTap={{ scale: 0.97 }}
   474→              suppressHydrationWarning
   475→            >
   476→              <div className="flex items-center justify-between">
   477→                <div className="flex items-center gap-3">
   478→                  <span className="text-xl">{svc.icon}</span>
   479→                  <span className={`text-sm font-medium ${selected ? "text-gray-900" : "text-gray-700"}`}>
   480→                    {svc.label}

   482→                </div>
   483→                {selected && (
   484→                  <motion.span
   485→                    initial={{ scale: 0 }}
   486→                    animate={{ scale: 1 }}
   487→                    transition={{ type: "spring", stiffness: 400, damping: 20 }}
   488→                    className="text-gray-900 font-bold text-sm"
   489→                  >
   490→                    ✓
   491→                  </motion.span>
   492→                )}
   493→              </div>
   494→            </motion.button>
   495→          )
   496→        })}
   497→      </div>
   498→    </div>
   499→  )
   500→}
