   109→// ─── Helpers ────────────────────────────────────────────────────────────────
   110→
   111→function getPainLimit(serviceCount: number): number {
   112→  if (serviceCount === 1) return 2
   113→  if (serviceCount === 2) return 3
   114→  return 4
   115→}
   116→
   117→function getAvailablePains(services: string[]): string[] {
   118→  const pains = new Set<string>()
   119→  for (const svc of services) {
   120→    const entry = SERVICE_PAIN_MAP[svc]
   121→    if (entry) entry.painPoints.forEach((p) => pains.add(p))
   122→  }
   123→  return [...pains]
   124→}
   125→
   126→function getAvailableProjects(services: string[]): string[] {
   127→  const projects = new Set<string>()
   128→  for (const svc of services) {
   129→    const entry = SERVICE_PAIN_MAP[svc]
   130→    if (entry) entry.projectDetails.forEach((p) => projects.add(p))
   131→  }
   132→  return [...projects]
   133→}

   137→const STEP_VARIANTS = {
   138→  enter: (dir: number) => ({

   143→  exit: (dir: number) => ({

   145→    opacity: 0,

   147→}

   151→export default function GetStartedPage() {

   153→  const mousePosition = useMousePosition(pageRef)
   154→
   155→  const [state, setState] = useState<QuestionnaireState>(INITIAL_STATE)
   156→  const [direction, setDirection] = useState(1)
   157→  const [submitting, setSubmitting] = useState(false)

   159→
   160→  const update = useCallback(<K extends keyof QuestionnaireState>(key: K, value: QuestionnaireState[K]) => {
   161→    setState((prev) => ({ ...prev, [key]: value }))
   162→  }, [])
   163→
   164→  const goNext = useCallback(() => {
   165→    setDirection(1)
   166→    setState((prev) => ({ ...prev, step: Math.min(prev.step + 1, 7) as Step }))
   167→  }, [])
   168→
   169→  const submitLead = useCallback(async () => {
   170→    setSubmitting(true)
   171→    setSubmitError("")
   172→    try {
   173→      const res = await fetch("/api/leads", {
   174→        method: "POST",
   175→        headers: { "Content-Type": "application/json" },
   176→        body: JSON.stringify({
   177→          firstName: state.firstName,
   178→          lastName: state.lastName,
   179→          email: state.email,
   180→          phone: state.phone,
   181→          company: state.company,
   182→          services: state.selectedServices,
   183→          painPoints: state.selectedPains.filter((p) => p !== "__custom__"),
   184→          painCustomText: state.painCustomText,
   185→          projectDetails: state.selectedProjects.filter((p) => p !== "__custom_project__"),
   186→          projectCustomText: state.projectCustomText,
   187→          contactPreferences: state.contactPrefs,
   188→          bestTime: state.bestTime,
   189→          budget: state.budget,
   190→          additionalNotes: state.additionalNotes,
   191→        }),
   192→      })
   193→      const data = await res.json()
   194→      if (!res.ok) {
   195→        setSubmitError(data.error || "Something went wrong. Please try again.")
   196→        setSubmitting(false)
   197→        return
   198→      }
   199→      // Success — advance to step 7
   200→      setDirection(1)
   201→      setState((prev) => ({ ...prev, step: 7 as Step }))
   202→    } catch {
   203→      setSubmitError("Network error. Please check your connection and try again.")
   204→    } finally {
   205→      setSubmitting(false)
   206→    }
   207→  }, [state])
   208→
   209→  const goBack = useCallback(() => {
   210→    setDirection(-1)
   211→    setState((prev) => ({ ...prev, step: Math.max(prev.step - 1, 1) as Step }))
   212→  }, [])
   213→
   214→  const goToStep = useCallback((step: Step) => {
   215→    setDirection(step > state.step ? 1 : -1)
   216→    setState((prev) => ({ ...prev, step }))
   217→  }, [state.step])
   218→
   219→  const toggleService = useCallback((svc: string) => {
   220→    setState((prev) => {
   221→      const has = prev.selectedServices.includes(svc)
   222→      const next = has
   223→        ? prev.selectedServices.filter((s) => s !== svc)
   224→        : [...prev.selectedServices, svc]
   225→      // Reset downstream state
   226→      return { ...prev, selectedServices: next, selectedPains: [], painCustomText: "", selectedProjects: [], projectCustomText: "" }
   227→    })
   228→  }, [])
   229→
   230→  const togglePain = useCallback((pain: string) => {
   231→    setState((prev) => {
   232→      if (pain === "__custom__") {
   233→        const has = prev.selectedPains.includes("__custom__")
   234→        if (!has && prev.selectedPains.length >= getPainLimit(prev.selectedServices.length)) return prev
   235→        const next = has
   236→          ? prev.selectedPains.filter((p) => p !== "__custom__")
   237→          : [...prev.selectedPains, "__custom__"]
   238→        return { ...prev, selectedPains: next }
   239→      }
   240→      const has = prev.selectedPains.includes(pain)
   241→      if (!has && prev.selectedPains.length >= getPainLimit(prev.selectedServices.length)) return prev
   242→      const next = has
   243→        ? prev.selectedPains.filter((p) => p !== pain)
   244→        : [...prev.selectedPains, pain]
   245→      return { ...prev, selectedPains: next }
   246→    })
   247→  }, [])
   248→
   249→  const toggleProject = useCallback((proj: string) => {
   250→    setState((prev) => {
   251→      if (proj === "__custom_project__") {
   252→        const has = prev.selectedProjects.includes("__custom_project__")
   253→        if (!has && prev.selectedProjects.length >= 3) return prev
   254→        const next = has
   255→          ? prev.selectedProjects.filter((p) => p !== "__custom_project__")
   256→          : [...prev.selectedProjects, "__custom_project__"]
   257→        return { ...prev, selectedProjects: next }
   258→      }
   259→      const has = prev.selectedProjects.includes(proj)
   260→      if (!has && prev.selectedProjects.length >= 3) return prev
   261→      const next = has
   262→        ? prev.selectedProjects.filter((p) => p !== proj)
   263→        : [...prev.selectedProjects, proj]
   264→      return { ...prev, selectedProjects: next }
   265→    })
   266→  }, [])
   267→
   268→  const toggleContactPref = useCallback((pref: string) => {
   269→    setState((prev) => {
   270→      const has = prev.contactPrefs.includes(pref)
   271→      return { ...prev, contactPrefs: has ? prev.contactPrefs.filter((p) => p !== pref) : [...prev.contactPrefs, pref] }
   272→    })
   273→  }, [])
   274→
   275→  // Derived values
   276→  const painLimit = getPainLimit(state.selectedServices.length)
   277→  const availablePains = useMemo(() => getAvailablePains(state.selectedServices), [state.selectedServices])
   278→  const availableProjects = useMemo(() => getAvailableProjects(state.selectedServices), [state.selectedServices])
   279→  const selectedBudget = BUDGET_OPTIONS.find((b) => b.value === state.budget)
   280→  const totalSteps = 6
   281→  const progressPct = state.step <= 6 ? (state.step / totalSteps) * 100 : 100
   282→
   283→  // Validation helpers
   284→  const canStep1 = state.selectedServices.length >= 1
   285→  const canStep2 = state.selectedPains.length >= 1
   286→  const canStep3 = state.selectedProjects.length >= 1
   287→  const canStep4 = state.firstName.trim() !== "" && state.email.trim() !== "" && state.contactPrefs.length >= 1
   288→  const canStep5 = state.budget !== ""
   289→
   290→  const canProceed = state.step === 1 ? canStep1
   291→    : state.step === 2 ? canStep2
   292→    : state.step === 3 ? canStep3
   293→    : state.step === 4 ? canStep4
   294→    : state.step === 5 ? canStep5
   295→    : true
   296→
   297→  // ─── Render ─────────────────────────────────────────────────────────────────
   298→
   299→  return (
   300→    <main ref={pageRef} className="relative min-h-screen bg-white">
   301→      {/* Step 7 uses its own dark background, steps 1-6 use circuitry */}
   302→      {state.step <= 6 && (
   303→        <CircuitryBackground mousePosition={mousePosition} />
   304→      )}
   305→
   306→      {/* ── Progress Bar (steps 1-6 only) ── */}
   307→      {state.step <= 6 && (
   308→        <div className="fixed top-0 left-0 right-0 z-50 h-[2px] bg-gray-100">
   309→          <motion.div
   310→            className="h-full bg-[#111111]"
   311→            initial={false}
   312→            animate={{ width: `${progressPct}%` }}
   313→            transition={{ duration: 0.4, ease: EASE }}
   314→          />
   315→        </div>
   316→      )}
   317→
   318→      {/* ── Hero (compact, steps 1-6 only) ── */}
   319→      {state.step <= 6 && (
   320→        <section className="relative min-h-[40vh] flex items-center justify-center px-4 pt-8 pb-4">
   321→          <div className="text-center max-w-3xl mx-auto relative z-10">
   322→            <motion.span
   323→              className="inline-block px-4 py-1.5 bg-gray-100 text-gray-600 text-xs font-medium tracking-wider uppercase rounded-full mb-6 border border-gray-200"
   324→              initial={{ scale: 0.8, opacity: 0 }}
   325→              animate={{ scale: 1, opacity: 1 }}
   326→              transition={{ duration: 0.5, ease: EASE }}
   327→            >
   328→              Get Started
   329→            </motion.span>
   330→            <motion.h1
   331→              className="text-4xl md:text-6xl font-bold tracking-tight leading-tight text-gray-900"
   332→              style={{ fontFamily: "var(--font-space-grotesk)" }}
   333→              initial={{ y: 30, opacity: 0 }}
   334→              animate={{ y: 0, opacity: 1 }}
   335→              transition={{ duration: 0.7, ease: EASE }}
   336→            >
   337→              Let&apos;s Build Something
   338→            </motion.h1>
   339→            <motion.p
   340→              className="mt-6 text-lg text-gray-600 max-w-2xl mx-auto leading-relaxed"
   341→              initial={{ y: 20, opacity: 0 }}
   342→              animate={{ y: 0, opacity: 1 }}
   343→              transition={{ duration: 0.7, delay: 0.2, ease: EASE }}
   344→            >
   345→              A few quick questions so we can hit the ground running. Takes about 2 minutes.
   346→            </motion.p>
   347→          </div>
   348→        </section>
   349→      )}
   350→
   351→      {/* ── Step 7: Full dark receipt page ── */}
   352→      {state.step === 7 && (
   353→        <Step7 firstName={state.firstName} />
   354→      )}
   355→
   356→      {/* ── Steps 1-6: Questionnaire Area ── */}
   357→      {state.step <= 6 && (
   358→        <section className="relative z-10 pb-32 md:pb-20">
   359→          <div className="max-w-3xl mx-auto px-4">
   360→            <AnimatePresence mode="wait" custom={direction}>
   361→              <motion.div
   362→                key={state.step}
   363→                custom={direction}
   364→                variants={STEP_VARIANTS}
   365→                initial="enter"
   366→                animate="center"
   367→                exit="exit"
   368→                transition={{ duration: 0.4, ease: EASE }}

   422→          {/* ── Desktop Navigation (bottom) ── */}

   447→          {/* ── Mobile Navigation (fixed bottom bar) ── */}

   479→function Step1({
   480→  selectedServices,
   481→  toggleService,
   482→}: {
   483→  selectedServices: string[]
   484→  toggleService: (s: string) => void
   485→}) {
   486→  return (
   487→    <div>
   488→      <h2
   489→        className="text-2xl md:text-3xl font-bold text-gray-900 tracking-tight"
   490→        style={{ fontFamily: "var(--font-space-grotesk)" }}
   491→      >
   492→        What do you need help with?
   493→      </h2>
   494→      <p className="text-gray-500 text-sm mt-2 mb-2">Pick as many as you like.</p>
   495→      {selectedServices.length > 0 && (
   496→        <motion.p
   497→          initial={{ opacity: 0, y: 8 }}
   498→          animate={{ opacity: 1, y: 0 }}
   499→          className="text-gray-400 text-xs mb-4"
   500→        >
   501→          You picked {selectedServices.length} service{selectedServices.length !== 1 ? "s" : ""}
   502→        </motion.p>
   503→      )}
   504→
   505→      <div className="grid grid-cols-1 sm:grid-cols-2 gap-3 mt-6">
   506→        {Object.entries(SERVICE_PAIN_MAP).map(([key, svc]) => {
